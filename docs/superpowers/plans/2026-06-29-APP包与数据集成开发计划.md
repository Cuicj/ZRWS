```java
package com.zrws.approval.service;

import com.alibaba.excel.EasyExcel;
import com.zrws.approval.domain.entity.*;
import com.zrws.approval.mapper.*;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.jdbc.core.JdbcTemplate;
import org.springframework.stereotype.Service;

import java.io.File;
import java.util.*;

@Slf4j
@Service
public class DataExportService {

    @Autowired
    private BoDefinitionMapper boDefinitionMapper;
    @Autowired
    private BoFieldMapper boFieldMapper;
    @Autowired
    private ExportTaskMapper exportTaskMapper;
    @Autowired
    private JdbcTemplate jdbcTemplate;

    public ExportTask createExportTask(String boCode, String fileFormat, 
                                        String filterConditions, String fieldList) {
        ExportTask task = new ExportTask();
        task.setTaskNo("EXP" + System.currentTimeMillis());
        task.setBoCode(boCode);
        task.setFileFormat(fileFormat);
        task.setFilterConditions(filterConditions);
        task.setFieldList(fieldList);
        task.setStatus("PENDING");
        task.setCreatedTime(java.time.LocalDateTime.now());
        exportTaskMapper.insert(task);
        return task;
    }

    public List<Map<String, Object>> queryExportData(String boCode, 
            String filterConditions, List<String> fieldList) {
        BoDefinition bo = boDefinitionMapper.selectByBoCode(boCode);
        if (bo == null) return Collections.emptyList();
        
        List<BoField> fields = boFieldMapper.selectByBoId(bo.getBoId());
        String columns = buildColumns(fields, fieldList);
        String sql = "SELECT " + columns + " FROM " + bo.getTargetTable();
        if (filterConditions != null && !filterConditions.isEmpty()) {
            sql += " WHERE " + filterConditions;
        }
        return jdbcTemplate.queryForList(sql);
    }

    public String exportToExcel(String boCode, List<Map<String, Object>> data) {
        BoDefinition bo = boDefinitionMapper.selectByBoCode(boCode);
        List<BoField> fields = boFieldMapper.selectByBoId(bo.getBoId());
        
        String outputPath = System.getProperty("java.io.tmpdir") + "/export/";
        new File(outputPath).mkdirs();
        String fileName = bo.getBoName() + "_" + System.currentTimeMillis() + ".xlsx";
        String fullPath = outputPath + fileName;
        
        List<List<String>> head = new ArrayList<>();
        List<List<Object>> rows = new ArrayList<>();
        
        for (BoField f : fields) {
            head.add(Collections.singletonList(f.getFieldName()));
        }
        for (Map<String, Object> row : data) {
            List<Object> rowData = new ArrayList<>();
            for (BoField f : fields) {
                rowData.add(row.get(f.getFieldCode()));
            }
            rows.add(rowData);
        }
        
        EasyExcel.write(fullPath).head(head).sheet("数据").doWrite(rows);
        return fullPath;
    }

    private String buildColumns(List<BoField> fields, List<String> fieldList) {
        if (fieldList == null || fieldList.isEmpty()) {
            return "*";
        }
        return String.join(", ", fieldList);
    }
}
```

```java
package com.zrws.approval.controller;

import com.zrws.approval.domain.entity.ExportTask;
import com.zrws.approval.service.DataExportService;
import com.zrws.approval.mapper.ExportTaskMapper;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.core.io.FileSystemResource;
import org.springframework.http.HttpHeaders;
import org.springframework.http.MediaType;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;

import java.io.File;
import java.net.URLEncoder;
import java.util.*;

@Slf4j
@RestController
@RequestMapping("/api/v1/export")
@CrossOrigin(origins = "*")
public class DataExportController {

    @Autowired
    private DataExportService dataExportService;
    @Autowired
    private ExportTaskMapper exportTaskMapper;

    @PostMapping("/create")
    public ResponseEntity<Map<String, Object>> createExportTask(
            @RequestBody Map<String, Object> request) {
        String boCode = (String) request.get("boCode");
        String fileFormat = (String) request.getOrDefault("fileFormat", "xlsx");
        @SuppressWarnings("unchecked")
        Map<String, Object> filters = (Map<String, Object>) request.get("filters");
        @SuppressWarnings("unchecked")
        List<String> fields = (List<String>) request.get("fields");
        
        ExportTask task = dataExportService.createExportTask(
            boCode, fileFormat, 
            filters != null ? filters.toString() : null,
            fields != null ? fields.toString() : null
        );
        
        Map<String, Object> result = new HashMap<>();
        result.put("success", true);
        result.put("task", task);
        return ResponseEntity.ok(result);
    }

    @PostMapping("/execute/{taskId}")
    public ResponseEntity<Map<String, Object>> executeExport(@PathVariable Long taskId) {
        ExportTask task = exportTaskMapper.selectById(taskId);
        if (task == null) {
            return ResponseEntity.badRequest().body(
                Collections.singletonMap("error", "任务不存在"));
        }
        
        task.setStatus("PROCESSING");
        task.setStartTime(java.time.LocalDateTime.now());
        exportTaskMapper.updateById(task);
        
        try {
            List<Map<String, Object>> data = dataExportService.queryExportData(
                task.getBoCode(), task.getFilterConditions(), null);
            
            String filePath = dataExportService.exportToExcel(task.getBoCode(), data);
            File file = new File(filePath);
            
            task.setFilePath(filePath);
            task.setFileName(file.getName());
            task.setFileSize(file.length());
            task.setTotalRows(data.size());
            task.setStatus("SUCCESS");
            task.setEndTime(java.time.LocalDateTime.now());
            exportTaskMapper.updateById(task);
            
            Map<String, Object> result = new HashMap<>();
            result.put("success", true);
            result.put("task", task);
            return ResponseEntity.ok(result);
        } catch (Exception e) {
            log.error("导出失败", e);
            task.setStatus("FAILED");
            task.setErrorMessage(e.getMessage());
            exportTaskMapper.updateById(task);
            return ResponseEntity.badRequest().body(
                Collections.singletonMap("error", e.getMessage()));
        }
    }

    @GetMapping("/download/{taskId}")
    public ResponseEntity<FileSystemResource> downloadFile(@PathVariable Long taskId) {
        ExportTask task = exportTaskMapper.selectById(taskId);
        if (task == null || !"SUCCESS".equals(task.getStatus())) {
            return ResponseEntity.notFound().build();
        }
        
        File file = new File(task.getFilePath());
        if (!file.exists()) {
            return ResponseEntity.notFound().build();
        }
        
        try {
            String encodedFileName = URLEncoder.encode(task.getFileName(), "UTF-8");
            return ResponseEntity.ok()
                .header(HttpHeaders.CONTENT_DISPOSITION, 
                    "attachment; filename=\"" + encodedFileName + "\"")
                .contentType(MediaType.APPLICATION_OCTET_STREAM)
                .contentLength(file.length())
                .body(new FileSystemResource(file));
        } catch (Exception e) {
            return ResponseEntity.badRequest().build();
        }
    }

    @GetMapping("/tasks")
    public ResponseEntity<Map<String, Object>> listTasks(
            @RequestParam(required = false) String boCode) {
        List<ExportTask> tasks;
        if (boCode != null) {
            tasks = exportTaskMapper.selectList(
                new com.baomidou.mybatisplus.core.conditions.query.QueryWrapper<ExportTask>()
                    .eq("bo_code", boCode).orderByDesc("created_time"));
        } else {
            tasks = exportTaskMapper.selectList(
                new com.baomidou.mybatisplus.core.conditions.query.QueryWrapper<ExportTask>()
                    .orderByDesc("created_time").last("LIMIT 50"));
        }
        
        Map<String, Object> result = new HashMap<>();
        result.put("success", true);
        result.put("tasks", tasks);
        return ResponseEntity.ok(result);
    }
}
```

```java
package com.zrws.approval.service;

import com.itextpdf.kernel.pdf.*;
import com.itextpdf.kernel.pdf.stamp.PdfStamp;
import com.itextpdf.layout.Document;
import com.itextpdf.layout.element.Image;
import com.itextpdf.io.image.ImageDataFactory;
import com.zrws.approval.domain.entity.SealConfig;
import com.zrws.approval.mapper.SealConfigMapper;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.io.ByteArrayOutputStream;
import java.io.FileInputStream;
import java.security.KeyStore;
import java.security.PrivateKey;
import java.security.cert.Certificate;
import java.util.List;

@Slf4j
@Service
public class ElectronicSealService {

    @Autowired
    private SealConfigMapper sealConfigMapper;

    public byte[] addSealToPdf(String pdfPath, Long sealId, 
            float x, float y, int pageNum) throws Exception {
        SealConfig seal = sealConfigMapper.selectById(sealId);
        if (seal == null) throw new RuntimeException("签章不存在");
        
        ByteArrayOutputStream baos = new ByteArrayOutputStream();
        PdfReader reader = new PdfReader(pdfPath);
        PdfWriter writer = new PdfWriter(baos);
        PdfDocument pdfDoc = new PdfDocument(reader, writer);
        Document doc = new Document(pdfDoc);
        
        if (seal.getSealImage() != null) {
            Image sealImage = new Image(
                ImageDataFactory.create(seal.getSealImage()));
            sealImage.setFixedPosition(pageNum, x, y);
            sealImage.scaleAbsolute(100, 100);
            doc.add(sealImage);
        }
        
        doc.close();
        return baos.toByteArray();
    }

    public byte[] signPdf(byte[] pdfData, Long sealId) throws Exception {
        SealConfig seal = sealConfigMapper.selectById(sealId);
        if (seal == null) throw new RuntimeException("签章配置不存在");
        if (seal.getCertificatePath() == null) throw new RuntimeException("证书未配置");
        
        KeyStore ks = KeyStore.getInstance("PKCS12");
        ks.load(new FileInputStream(seal.getCertificatePath()),
                seal.getCertificatePassword().toCharArray());
        String alias = ks.aliases().nextElement();
        PrivateKey privateKey = (PrivateKey) ks.getKey(alias,
                seal.getCertificatePassword().toCharArray());
        Certificate[] chain = ks.getCertificateChain(alias);
        
        PdfReader reader = new PdfReader(
            new java.io.ByteArrayInputStream(pdfData));
        ByteArrayOutputStream baos = new ByteArrayOutputStream();
        
        PdfSigner signer = new PdfSigner(reader, baos, new StampingProperties());
        PdfSignatureAppearance appearance = signer.getSignatureAppearance();
        appearance.setReason("智壤卫士-数据认证");
        appearance.setLocation("智壤卫士平台");
        
        signer.signDetached(new BouncyCastleDigest(), 
            new PrivateKeySignature(privateKey, "SHA256", "BC"),
            chain, null, null, null, 0, 
            PdfSigner.CryptoStandard.CMS);
        
        return baos.toByteArray();
    }
```

```java
package com.zrws.approval.service;

import com.zrws.approval.domain.entity.ApiKey;
import com.zrws.approval.mapper.ApiKeyMapper;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.data.redis.core.StringRedisTemplate;
import org.springframework.stereotype.Service;

import javax.crypto.Mac;
import javax.crypto.spec.SecretKeySpec;
import java.nio.charset.StandardCharsets;
import java.util.Base64;
import java.util.concurrent.TimeUnit;

@Slf4j
@Service
public class OpenApiAuthService {

    @Autowired
    private ApiKeyMapper apiKeyMapper;
    
    @Autowired(required = false)
    private StringRedisTemplate redisTemplate;

    public boolean validateSignature(String apiKey, String timestamp,
                                     String signature, String method, String path, String body) {
        ApiKey key = apiKeyMapper.selectByApiKey(apiKey);
        if (key == null || !"ACTIVE".equals(key.getStatus())) return false;
        
        long ts = Long.parseLong(timestamp);
        if (Math.abs(System.currentTimeMillis() / 1000 - ts) > 300) return false;
        
        String stringToSign = method + "\n" + path + "\n" + timestamp + "\n" + 
                              (body != null ? body : "");
        String computedSign = hmacSHA256(key.getApiSecret(), stringToSign);
        
        if (computedSign.equals(signature)) {
            key.setRequestCount(key.getRequestCount() + 1);
            key.setLastRequestTime(java.time.LocalDateTime.now());
            apiKeyMapper.updateById(key);
            return true;
        }
        return false;
    }

    public boolean checkRateLimit(String apiKey) {
        if (redisTemplate == null) return true;
        String key = "openapi:ratelimit:" + apiKey;
        Long count = redisTemplate.opsForValue().increment(key);
        if (count == 1) {
            redisTemplate.expire(key, 60, TimeUnit.SECONDS);
        }
        return count <= 100;
    }

    private String hmacSHA256(String secret, String data) {
        try {
            Mac mac = Mac.getInstance("HmacSHA256");
            mac.init(new SecretKeySpec(secret.getBytes(StandardCharsets.UTF_8), "HmacSHA256"));
            byte[] hash = mac.doFinal(data.getBytes(StandardCharsets.UTF_8));
            return Base64.getEncoder().encodeToString(hash);
        } catch (Exception e) {
            throw new RuntimeException(e);
        }
    }

    public ApiKey generateApiKey(Long companyId, String companyName, String permissions) {
        ApiKey key = new ApiKey();
        key.setApiKey("ZRWS_" + generateRandomString(24));
        key.setApiSecret(generateRandomString(48));
        key.setCompanyId(companyId);
        key.setCompanyName(companyName);
        key.setPermissions(permissions);
        key.setStatus("ACTIVE");
        key.setRateLimit(100);
        key.setRequestCount(0);
        apiKeyMapper.insert(key);
        return key;
    }

    private String generateRandomString(int length) {
        String chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789";
        StringBuilder sb = new StringBuilder();
        java.util.Random r = new java.util.Random();
        for (int i = 0; i < length; i++) {
            sb.append(chars.charAt(r.nextInt(chars.length())));
        }
        return sb.toString();
    }
}
```

## 设计规范统一

所有新开发功能必须遵循以下设计规范：

### 颜色方案
- 主背景：`#FEFBF6` 暖米白（渐变到 `#F7F3ED`）
- 卡片背景：`#FAFAF8` → `#F5F2ED` 浅米色渐变
- 主色：`#C9A96E` 沙金色
- 主文字：`#5D4E37` 深棕色
- 次文字：`#8B7355` 中棕色
- 边框：`#E8E2D9` 浅米褐

### 组件规范
- 圆角：12-16px（24-32rpx移动端）
- 阴影：柔和棕色阴影，不使用黑色阴影
- 过渡：0.3s cubic-bezier(0.4, 0, 0.2, 1)
- Hover效果：卡片上浮2px + 阴影加深 + 边框变沙金色

### 交互规范
- 加载状态：骨架屏 + 渐入动画
- 空状态：图标 + 提示文字 + 操作按钮
- 错误状态：友好提示 + 重试按钮
- 下拉刷新：统一风格
