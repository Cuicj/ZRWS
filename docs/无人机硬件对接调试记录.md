### 2.1 SDK选择

| SDK类型 | 适用场景 | 支持平台 | 说明 |
|---------|---------|---------|------|
| Mobile SDK (MSDK) | 移动端App开发 | Android/iOS | 最常用，需配合遥控器 |
| Onboard SDK (OSDK) | 机载电脑开发 | Linux/ROS | 直接控制飞行控制器 |

**推荐组合**: MSDK 5.x + OSDK (用于高级功能)

#### 2.3.3 航点任务

```java
// 创建航点任务
public void createWaypointMission(List<Waypoint> waypoints) {
    FlightController fc = getFlightController();
    if (fc == null) return;
    
    // 构建航点任务
    WaypointMission.Builder builder = new WaypointMission.Builder();
    builder.waypointCount(waypoints.size())
           .autoFlightSpeed(8.0f)        // 自动飞行速度 m/s
           .maxFlightSpeed(15.0f)        // 最大飞行速度
           .flightPathMode(WaypointMission.FlightPathMode.NORMAL)
           .gotoFirstWaypointMode(WaypointMission.GotoFirstWaypointMode.SAFELY);
    
    // 添加航点
    for (Waypoint wp : waypoints) {
        Waypoint waypoint = new Waypoint(wp.lat, wp.lng, (float)wp.altitude);
        waypoint.speed = (float)wp.speed;
        waypoint.hoverTime = wp.hoverTime;
        
        // 设置航点动作
        if (wp.shootPhoto) {
            WaypointAction action = new WaypointAction(
                WaypointActionType.START_TAKE_PHOTO, 0
            );
            waypoint.addAction(action);
        }
        
        builder.addWaypoint(waypoint);
    }
    
    WaypointMission mission = builder.build();
    
    // 上传任务
    fc.uploadWaypointMission(mission, new CommonCallbacks.CompletionCallback() {
        @Override
        public void onResult(DJIError error) {
            if (error == null) {
                Log.i(TAG, "航点任务上传成功");
            } else {
                Log.e(TAG, "航点任务上传失败: " + error.getDescription());
            }
        }
    });
}
```

#### 3.2.2 认证流程

```java
// 1. 获取Access Token
public String getAccessToken(String appKey, String appSecret) {
    String url = "https://open.xag.cn/oauth/token";
    
    Map<String, String> params = new HashMap<>();
    params.put("grant_type", "client_credentials");
    params.put("client_id", appKey);
    params.put("client_secret", appSecret);
    
    HttpResponse response = HttpUtil.post(url, params);
    JSONObject json = JSON.parseObject(response.body());
    
    return json.getString("access_token");
}
```

#### 3.2.3 设备列表查询

```java
// 查询已绑定的设备
public List<XAGDevice> getDeviceList(String accessToken) {
    String url = "https://open.xag.cn/v1/devices";
    
    Map<String, String> headers = new HashMap<>();
    headers.put("Authorization", "Bearer " + accessToken);
    
    HttpResponse response = HttpUtil.get(url, headers);
    JSONObject json = JSON.parseObject(response.body());
    
    List<XAGDevice> devices = new ArrayList<>();
    JSONArray array = json.getJSONArray("data");
    for (int i = 0; i < array.size(); i++) {
        JSONObject obj = array.getJSONObject(i);
        XAGDevice device = new XAGDevice();
        device.setDeviceId(obj.getString("device_id"));
        device.setSerialNumber(obj.getString("sn"));
        device.setModel(obj.getString("model"));
        device.setOnline(obj.getBoolean("online"));
        devices.add(device);
    }
    
    return devices;
}
```

#### 3.2.4 实时遥测数据

极飞通过WebSocket推送实时数据:

```java
// 建立WebSocket连接
public void subscribeTelemetry(String accessToken, String deviceId) {
    String wsUrl = "wss://open.xag.cn/v1/devices/" + deviceId + "/telemetry";
    
    WebSocketClient client = new WebSocketClient(new URI(wsUrl)) {
        @Override
        public void onOpen(ServerHandshake handshakedata) {
            // 发送认证信息
            String authMsg = "{\"access_token\":\"" + accessToken + "\"}";
            send(authMsg);
        }
        
        @Override
        public void onMessage(String message) {
            JSONObject json = JSON.parseObject(message);
            processTelemetry(json);
        }
        
        @Override
        public void onClose(int code, String reason, boolean remote) {
            Log.i(TAG, "WebSocket连接关闭");
        }
        
        @Override
        public void onError(Exception ex) {
            Log.e(TAG, "WebSocket错误", ex);
        }
    };
    
    client.connect();
}

// 处理遥测数据
private void processTelemetry(JSONObject json) {
    TelemetryData data = new TelemetryData();
    data.setDeviceId(json.getString("device_id"));
    data.setLatitude(json.getDouble("lat"));
    data.setLongitude(json.getDouble("lng"));
    data.setAltitude(json.getDouble("height"));
    data.setBattery(json.getInteger("battery"));
    data.setSpeed(json.getDouble("speed"));
    data.setFlying(json.getBoolean("flying"));
    
    // 发送到服务器
    sendToServer(data);
}
```

### 3.3 作业任务下发

#### 3.3.1 创建农业作业任务

```java
// 创建喷洒作业任务
public String createSprayingTask(String accessToken, SprayingPlan plan) {
    String url = "https://open.xag.cn/v1/flight/spraying-plans";
    
    Map<String, String> headers = new HashMap<>();
    headers.put("Authorization", "Bearer " + accessToken);
    headers.put("Content-Type", "application/json");
    
    JSONObject body = new JSONObject();
    body.put("device_id", plan.getDeviceId());
    body.put("device_sn", plan.getDeviceSn());
    
    // 航线数据
    JSONObject route = new JSONObject();
    route.put("type", "polygon");
    route.put("height", plan.getFlightHeight());
    route.put("speed", plan.getFlightSpeed());
    
    // 区域坐标
    JSONArray coordinates = new JSONArray();
    for (Coordinate coord : plan.getPolygon()) {
        coordinates.add(new double[]{coord.getLng(), coord.getLat()});
    }
    route.put("coordinates", coordinates);
    body.put("route", route);
    
    // 作业参数
    JSONObject settings = new JSONObject();
    settings.put("spray_rate", plan.getSprayRate());    // ml/亩
    settings.put("flow_rate", plan.getFlowRate());      // L/min
    settings.put("optical_flow", plan.getOpticalFlow());// 流量控制
    body.put("settings", settings);
    
    HttpResponse response = HttpUtil.post(url, headers, body.toJSONString());
    JSONObject result = JSON.parseObject(response.body());
    
    return result.getString("plan_id");
}
```

## 5. 常见问题与解决方案

### 5.1 大疆无人机

| 问题 | 原因 | 解决方案 |
|------|------|---------|
| SDK注册失败 | App Key错误 | 检查DJI Developer平台的App Key |
| 产品无法连接 | USB调试未开启 | 在遥控器设置中开启USB调试 |
| 航点任务上传失败 | 航点数量超限 | M350最多10000个航点 |
| 任务执行中断 | 电池电量不足 | 设置30%电量返航 |
| GPS精度差 | 卫星数不足 | 等待卫星数>15再起飞 |
| RTK定位失败 | 基站连接断开 | 检查基站网络和电源 |

### 5.2 极飞无人机

| 问题 | 原因 | 解决方案 |
|------|------|---------|
| API返回401 | Token过期 | 刷新Access Token |
| 设备不在线 | 未连接极飞智图 | 检查设备网络状态 |
| 作业任务创建失败 | 区域坐标错误 | 检查KML格式 |
| WebSocket断开 | 网络不稳定 | 实现自动重连机制 |
| 设备绑定失败 | 设备已被其他账号绑定 | 联系极飞技术支持 |
| 遥测数据延迟 | 服务器负载高 | 降低数据推送频率 |

## 7. 配置文件示例

### 7.1 application.yml

```yaml
# 无人机SDK配置
drone:
  # 大疆配置
  dji:
    app-key: ${DJI_APP_KEY:your-dji-app-key}
    app-secret: ${DJI_APP_SECRET:your-dji-app-secret}
    debug-mode: true
    
  # 极飞配置
  xag:
    api-key: ${XAG_API_KEY:your-xag-api-key}
    api-secret: ${XAG_API_SECRET:your-xag-secret}
    base-url: https://open.xag.cn
    ws-url: wss://open.xag.cn
    
  # 遥测数据配置
  telemetry:
    push-interval: 1000    # 推送间隔(ms)
    buffer-size: 100      # 缓冲区大小
    enable-compress: true  # 启用压缩
```

### 7.2 环境变量

```bash
# 大疆
export DJI_APP_KEY=your-dji-app-key
export DJI_APP_SECRET=your-dji-app-secret

# 极飞
export XAG_API_KEY=your-xag-api-key
export XAG_API_SECRET=your-xag-secret
```
